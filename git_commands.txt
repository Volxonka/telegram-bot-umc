# Команды для загрузки в GitHub через терминал

# 1. Инициализация Git репозитория
git init

# 2. Добавление удаленного репозитория
git remote add origin https://github.com/Volxonka/telegram-bot-umc.git

# 3. Добавление всех файлов
git add .

# 4. Коммит изменений
git commit -m "Initial commit: Telegram bot for UMC"

# 5. Отправка в GitHub
git push -u origin main

# Если возникнет ошибка с main, попробуй:
git branch -M main
git push -u origin main

# Если нужно настроить Git впервые:
git config --global user.name "Volxonka"
git config --global user.email "your-email@example.com"
